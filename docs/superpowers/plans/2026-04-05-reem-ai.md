> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Reem AI, a conversational AI agent page in the dashboard that can query business data, create commissions/paiements (with confirmation), search Google Drive, draft emails, and have free-form conversations.

**Architecture:** New page `/dashboard/agent` with a chat interface. API route `/api/agent/chat` uses Claude Sonnet with function calling (tools) to query Supabase and Drive. A separate `/api/agent/confirm` route handles confirmed creates. Messages persisted per-user in `agent_messages` table.

**Tech Stack:** Next.js 16, TypeScript, Supabase, Anthropic SDK (Claude Sonnet), Zod, Tailwind

// Execute a tool and return results
interface ToolInput {
  client_id?: string
  prime_id?: string
  status?: string
  mois?: string
  limit?: number
  ca?: number
  commission?: number
  dossiers?: number
  date?: string
  montant?: number
  label?: string
  query?: string
  to?: string
  subject?: string
  body?: string
}

export async function executeAgentTool(
  toolName: string,
  input: ToolInput,
  context: { clientId: string | null; userId: string; googleAccessToken: string | null }
): Promise<{ type: 'data' | 'confirm' | 'draft_email'; result: unknown }> {
  const effectiveClientId = input.client_id ?? context.clientId

  switch (toolName) {
    case 'query_commissions': {
      let query = supabaseAdmin
        .from('commissions')
        .select('*, prime:primes(name, color, icon)')
        .order('created_at', { ascending: false })
        .limit(20)
      if (effectiveClientId) query = query.eq('client_id', effectiveClientId)
      if (input.prime_id) query = query.eq('prime_id', input.prime_id)
      if (input.status) query = query.eq('status', input.status)
      if (input.mois) query = query.eq('mois', input.mois)
      const { data, error } = await query
      if (error) throw new Error(error.message)
      return { type: 'data', result: data }
    }

    case 'query_paiements': {
      let query = supabaseAdmin
        .from('paiements')
        .select('*')
        .order('date', { ascending: false })
        .limit(20)
      if (effectiveClientId) query = query.eq('client_id', effectiveClientId)
      if (input.status) query = query.eq('status', input.status)
      const { data, error } = await query
      if (error) throw new Error(error.message)
      return { type: 'data', result: data }
    }

    case 'query_clients': {
      const { data, error } = await supabaseAdmin
        .from('clients')
        .select('*')
        .order('name')
      if (error) throw new Error(error.message)
      return { type: 'data', result: data }
    }

    case 'query_sommes_dues': {
      let query = supabaseAdmin
        .from('sommes_dues')
        .select('*')
        .order('created_at', { ascending: false })
      if (effectiveClientId) query = query.eq('client_id', effectiveClientId)
      const { data, error } = await query
      if (error) throw new Error(error.message)
      return { type: 'data', result: data }
    }

    case 'query_activity': {
      const { data, error } = await supabaseAdmin
        .from('activity_log')
        .select('*, user:users(display_name)')
        .order('created_at', { ascending: false })
        .limit(input.limit ?? 10)
      if (error) throw new Error(error.message)
      return { type: 'data', result: data }
    }

    case 'create_commission': {
      // Fetch prime name for display
      const { data: prime } = await supabaseAdmin
        .from('primes')
        .select('name, icon')
        .eq('id', input.prime_id ?? '')
        .single()
      return {
        type: 'confirm',
        result: {
          action: 'create_commission',
          data: {
            prime_id: input.prime_id,
            prime_name: prime ? `${prime.icon} ${prime.name}` : input.prime_id,
            ca: input.ca,
            commission: input.commission,
            dossiers: input.dossiers ?? 0,
            mois: input.mois,
            status: input.status ?? 'due',
            client_id: effectiveClientId,
            user_id: context.userId,
            created_by: context.userId,
          },
        },
      }
    }

    case 'create_paiement': {
      return {
        type: 'confirm',
        result: {
          action: 'create_paiement',
          data: {
            date: input.date,
            montant: input.montant,
            label: input.label,
            status: input.status ?? 'en_attente',
            client_id: effectiveClientId,
            created_by: context.userId,
          },
        },
      }
    }

    case 'search_drive': {
      if (!context.googleAccessToken) {
        return { type: 'data', result: { error: 'Google Drive non connecté. Connectez-vous depuis la page Drive.' } }
      }
      const q = `name contains '${(input.query ?? '').replace(/'/g, "\\'")}' and trashed = false`
      const params = new URLSearchParams({
        q,
        fields: 'files(id,name,mimeType,modifiedTime,size)',
        pageSize: '10',
      })
      const res = await fetch(`https://www.googleapis.com/drive/v3/files?${params}`, {
        headers: { Authorization: `Bearer ${context.googleAccessToken}` },
      })
      if (!res.ok) return { type: 'data', result: { error: 'Erreur Drive' } }
      const data = await res.json()
      return { type: 'data', result: data.files ?? [] }
    }

    case 'draft_email': {
      return {
        type: 'draft_email',
        result: { to: input.to, subject: input.subject, body: input.body },
      }
    }

    default:
      return { type: 'data', result: { error: `Tool inconnu: ${toolName}` } }
  }
}

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { z } from 'zod'
import { getSessionUser } from '@/lib/auth'
import { supabaseAdmin } from '@/lib/supabase'
import Anthropic from '@anthropic-ai/sdk'
import { AGENT_TOOLS, executeAgentTool } from '@/lib/agent-tools'

const SYSTEM_PROMPT = `Tu es Reem AI, l'assistante intelligente de LR Consulting W.L.L, basée au Royaume de Bahreïn.
Tu parles toujours en français.
Tu peux :
- Interroger les données de commissions, paiements, clients, sommes dues, et l'historique d'activité
- Créer des commissions et des paiements (tu dois toujours proposer une confirmation avant de créer)
- Chercher des fichiers dans Google Drive
- Composer des brouillons d'emails
- Discuter librement de tout sujet

Quand tu présentes des données chiffrées, utilise le format français (espaces pour les milliers, € pour les euros).
Quand tu proposes de créer une commission, vérifie que la prime existe dans les données.
Sois concise, professionnelle et utile.`

const messageSchema = z.object({
  message: z.string().min(1).max(5000),
  clientId: z.string().nullable().optional(),
})

export async function POST(req: NextRequest) {
  const session = await getSessionUser()
  if (!session) return NextResponse.json({ error: 'Non authentifié.' }, { status: 401 })

  try {
    const body = await req.json()
    const parsed = messageSchema.safeParse(body)
    if (!parsed.success) {
      return NextResponse.json({ error: 'Données invalides.' }, { status: 400 })
    }

    const { message, clientId } = parsed.data
    const apiKey = process.env.ANTHROPIC_API_KEY
    if (!apiKey) {
      return NextResponse.json({ error: 'Clé API non configurée.' }, { status: 500 })
    }

    // Save user message
    await supabaseAdmin.from('agent_messages').insert({
      user_id: session.id,
      role: 'user',
      content: message,
    })

    // Load conversation history (last 30 messages)
    const { data: history } = await supabaseAdmin
      .from('agent_messages')
      .select('role, content, tool_data')
      .eq('user_id', session.id)
      .order('created_at', { ascending: true })
      .limit(30)

    const messages: Anthropic.MessageParam[] = (history ?? []).map(m => ({
      role: m.role as 'user' | 'assistant',
      content: m.content ?? '',
    }))

    // Get Google token for Drive search
    const googleRaw = req.cookies.get('google_tokens')?.value
    let googleAccessToken: string | null = null
    if (googleRaw) {
      try {
        const tokens = JSON.parse(googleRaw)
        if (tokens.access_token && Date.now() < tokens.expires_at) {
          googleAccessToken = tokens.access_token
        }
      } catch {
        // pas de token valide
      }
    }

    const client = new Anthropic({ apiKey })

    // Call Claude with tools — handle tool use loop
    let response = await client.messages.create({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 2000,
      system: SYSTEM_PROMPT,
      messages,
      tools: AGENT_TOOLS,
    })

    const toolResults: { type: string; result: unknown }[] = []

    // Tool use loop
    while (response.stop_reason === 'tool_use') {
      const toolUseBlocks = response.content.filter(b => b.type === 'tool_use')
      const toolResultContents: Anthropic.ToolResultBlockParam[] = []

      for (const block of toolUseBlocks) {
        if (block.type !== 'tool_use') continue
        try {
          const toolResult = await executeAgentTool(
            block.name,
            block.input as Record<string, unknown>,
            { clientId: clientId ?? null, userId: session.id, googleAccessToken }
          )
          toolResults.push(toolResult)
          toolResultContents.push({
            type: 'tool_result',
            tool_use_id: block.id,
            content: JSON.stringify(toolResult.result),
          })
        } catch (err) {
          toolResultContents.push({
            type: 'tool_result',
            tool_use_id: block.id,
            content: `Erreur: ${err instanceof Error ? err.message : 'Erreur inconnue'}`,
            is_error: true,
          })
        }
      }

      // Continue conversation with tool results
      messages.push({ role: 'assistant', content: response.content })
      messages.push({ role: 'user', content: toolResultContents })

      response = await client.messages.create({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 2000,
        system: SYSTEM_PROMPT,
        messages,
        tools: AGENT_TOOLS,
      })
    }

    // Extract final text
    const textBlock = response.content.find(b => b.type === 'text')
    const assistantText = textBlock && textBlock.type === 'text' ? textBlock.text : ''

    // Build tool_data from results
    const toolData = toolResults.length > 0 ? toolResults : null

    // Save assistant message
    await supabaseAdmin.from('agent_messages').insert({
      user_id: session.id,
      role: 'assistant',
      content: assistantText,
      tool_data: toolData,
    })

    return NextResponse.json({
      content: assistantText,
      tool_data: toolData,
    })
  } catch (err) {
    const errorMessage = err instanceof Error ? err.message : 'Erreur interne'
    return NextResponse.json({ error: errorMessage }, { status: 500 })
  }
}

import { NextRequest, NextResponse } from 'next/server'
import { z } from 'zod'
import { getSessionUser } from '@/lib/auth'
import { supabaseAdmin } from '@/lib/supabase'

const confirmSchema = z.object({
  action: z.enum(['create_commission', 'create_paiement']),
  data: z.record(z.string(), z.unknown()),
})

export async function POST(req: NextRequest) {
  const session = await getSessionUser()
  if (!session) return NextResponse.json({ error: 'Non authentifié.' }, { status: 401 })

  try {
    const body = await req.json()
    const parsed = confirmSchema.safeParse(body)
    if (!parsed.success) {
      return NextResponse.json({ error: 'Données invalides.' }, { status: 400 })
    }

    const { action, data } = parsed.data

    if (action === 'create_commission') {
      const { data: inserted, error } = await supabaseAdmin
        .from('commissions')
        .insert({
          prime_id: data.prime_id,
          ca: Number(data.ca),
          commission: Number(data.commission),
          dossiers: Number(data.dossiers) || 0,
          mois: data.mois,
          status: data.status ?? 'due',
          user_id: data.user_id ?? session.id,
          created_by: session.id,
          client_id: data.client_id,
        })
        .select('*')
        .single()
      if (error) return NextResponse.json({ error: error.message }, { status: 500 })

      await supabaseAdmin.from('activity_log').insert({
        user_id: session.id,
        action: 'create',
        entity_type: 'commission',
        entity_id: inserted.id,
        details: { description: `${session.display_name} a créé une commission via Reem AI` },
      })

      return NextResponse.json({ success: true, record: inserted })
    }

    if (action === 'create_paiement') {
      const { data: inserted, error } = await supabaseAdmin
        .from('paiements')
        .insert({
          date: data.date,
          montant: Number(data.montant),
          label: data.label,
          status: data.status ?? 'en_attente',
          created_by: session.id,
          client_id: data.client_id,
        })
        .select('*')
        .single()
      if (error) return NextResponse.json({ error: error.message }, { status: 500 })

      await supabaseAdmin.from('activity_log').insert({
        user_id: session.id,
        action: 'create',
        entity_type: 'paiement',
        entity_id: inserted.id,
        details: { description: `${session.display_name} a créé un paiement via Reem AI` },
      })

      return NextResponse.json({ success: true, record: inserted })
    }

    return NextResponse.json({ error: 'Action inconnue.' }, { status: 400 })
  } catch {
    return NextResponse.json({ error: 'Erreur serveur.' }, { status: 500 })
  }
}

'use client'

import { useState, useEffect, useCallback } from 'react'
import { supabase } from '@/lib/supabase'

interface AgentMessage {
  id: string
  user_id: string
  role: 'user' | 'assistant'
  content: string | null
  tool_data: { type: string; result: unknown }[] | null
  created_at: string
}

export function useAgentMessages(userId: string | undefined) {
  const [messages, setMessages] = useState<AgentMessage[]>([])
  const [loading, setLoading] = useState(true)
  const [sending, setSending] = useState(false)

  const load = useCallback(async () => {
    if (!userId) return
    setLoading(true)
    try {
      const { data, error } = await supabase
        .from('agent_messages')
        .select('*')
        .eq('user_id', userId)
        .order('created_at', { ascending: true })
      if (error) throw error
      setMessages((data ?? []) as AgentMessage[])
    } catch {
      // silencieux
    } finally {
      setLoading(false)
    }
  }, [userId])

  useEffect(() => { load() }, [load])

  const sendMessage = useCallback(async (
    message: string,
    clientId: string | null
  ): Promise<{ content: string; tool_data: { type: string; result: unknown }[] | null } | null> => {
    if (!userId || sending) return null
    setSending(true)

    // Optimistic user message
    const optimisticId = crypto.randomUUID()
    const userMsg: AgentMessage = {
      id: optimisticId,
      user_id: userId,
      role: 'user',
      content: message,
      tool_data: null,
      created_at: new Date().toISOString(),
    }
    setMessages(prev => [...prev, userMsg])

    try {
      const res = await fetch('/api/agent/chat', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ message, clientId }),
      })
      if (!res.ok) throw new Error('Erreur agent')
      const data = await res.json()

      // Reload to get server-generated IDs
      await load()
      return data
    } catch {
      // Remove optimistic message on error
      setMessages(prev => prev.filter(m => m.id !== optimisticId))
      return null
    } finally {
      setSending(false)
    }
  }, [userId, sending, load])

  const clearHistory = useCallback(async () => {
    try {
      const res = await fetch('/api/agent/clear', { method: 'DELETE' })
      if (res.ok) setMessages([])
    } catch {
      // silencieux
    }
  }, [])

  const confirmAction = useCallback(async (
    action: string,
    data: Record<string, unknown>
  ): Promise<boolean> => {
    try {
      const res = await fetch('/api/agent/confirm', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ action, data }),
      })
      return res.ok
    } catch {
      return false
    }
  }, [])

  return { messages, loading, sending, sendMessage, clearHistory, confirmAction }
}
```

'use client'

import { useState, type FormEvent } from 'react'

interface AgentInputProps {
  onSend: (message: string) => void
  disabled?: boolean
}

export default function AgentInput({ onSend, disabled }: AgentInputProps) {
  const [text, setText] = useState('')

This component renders a single message with support for text, data cards, confirm cards, and synthesis.

```typescript
'use client'

import { useState } from 'react'
import { useAuth } from '@/hooks/useAuth'
import { avatarInitials } from '@/lib/utils'

interface ToolData {
  type: string
  result: unknown
}

interface AgentMessageProps {
  role: 'user' | 'assistant'
  content: string | null
  toolData: ToolData[] | null
  onConfirm?: (action: string, data: Record<string, unknown>) => Promise<boolean>
}

function DataCard({ data }: { data: unknown }) {
  if (!data || typeof data !== 'object') return null
  const items = Array.isArray(data) ? data : [data]
  if (items.length === 0) return <p style={{ color: '#8898aa', fontSize: '12px', fontStyle: 'italic' }}>Aucun résultat</p>

  // Extract column names from first item
  const keys = Object.keys(items[0] as Record<string, unknown>).filter(k =>
    !['id', 'user_id', 'created_by', 'client_id', 'updated_at', 'created_at', 'prime', 'user'].includes(k)
  ).slice(0, 6)

  return (
    <div style={{ background: '#0f1117', border: '1px solid rgba(255,255,255,0.08)', borderRadius: '10px', overflow: 'hidden', marginTop: '6px' }}>
      <div style={{ padding: '8px 14px', borderBottom: '1px solid rgba(255,255,255,0.06)', display: 'flex', alignItems: 'center', gap: '8px', fontSize: '11px', fontWeight: 600, color: '#8898aa' }}>
        <span style={{ width: '6px', height: '6px', borderRadius: '50%', background: '#10b981' }} />
        {items.length} résultat{items.length > 1 ? 's' : ''}
      </div>
      <div style={{ overflowX: 'auto' }}>
        <table style={{ width: '100%', fontSize: '12px', borderCollapse: 'collapse' }}>
          <thead>
            <tr>
              {keys.map(k => (
                <th key={k} style={{ textAlign: 'left', padding: '8px 14px', fontSize: '9px', textTransform: 'uppercase', letterSpacing: '0.8px', color: '#3d4f63', fontWeight: 600 }}>
                  {k.replace(/_/g, ' ')}
                </th>
              ))}
            </tr>
          </thead>
          <tbody>
            {items.slice(0, 10).map((item, i) => (
              <tr key={i}>
                {keys.map(k => {
                  const val = (item as Record<string, unknown>)[k]
                  const display = val === null ? '—' : typeof val === 'object' ? JSON.stringify(val) : String(val)
                  return (
                    <td key={k} style={{ padding: '8px 14px', borderTop: '1px solid rgba(255,255,255,0.04)', color: '#e8edf5' }}>
                      {display}
                    </td>
                  )
                })}
              </tr>
            ))}
          </tbody>
        </table>
      </div>
    </div>
  )
}

function ConfirmCard({ data, onConfirm }: { data: { action: string; data: Record<string, unknown> }; onConfirm?: (action: string, data: Record<string, unknown>) => Promise<boolean> }) {
  const [confirmed, setConfirmed] = useState(false)
  const [confirming, setConfirming] = useState(false)

  async function handleConfirm() {
    if (!onConfirm || confirming) return
    setConfirming(true)
    const success = await onConfirm(data.action, data.data)
    if (success) setConfirmed(true)
    setConfirming(false)
  }

  const fields = Object.entries(data.data).filter(([k]) =>
    !['user_id', 'created_by', 'client_id', 'action'].includes(k)
  )

  return (
    <div style={{ background: '#0f1117', border: '1px solid rgba(245,158,11,0.25)', borderRadius: '10px', padding: '14px', marginTop: '6px' }}>
      <h4 style={{ fontSize: '12px', fontWeight: 600, color: confirmed ? '#10b981' : '#f59e0b', marginBottom: '10px', display: 'flex', alignItems: 'center', gap: '6px' }}>
        {confirmed ? '✓ Créé avec succès' : '⚠️ Confirmer la création'}
      </h4>
      <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr', gap: '8px', marginBottom: '12px' }}>
        {fields.map(([key, val]) => (
          <div key={key}>
            <label style={{ fontSize: '9px', textTransform: 'uppercase', letterSpacing: '0.5px', color: '#3d4f63' }}>
              {key.replace(/_/g, ' ')}
            </label>
            <p style={{ fontSize: '13px', color: '#e8edf5', fontWeight: 500, marginTop: '2px' }}>
              {typeof val === 'number' ? new Intl.NumberFormat('fr-FR').format(val) + (key.includes('ca') || key.includes('commission') || key.includes('montant') ? ' €' : '') : String(val ?? '—')}
            </p>
          </div>
        ))}
      </div>
      {!confirmed && (
        <div style={{ display: 'flex', gap: '8px', justifyContent: 'flex-end' }}>
          <button
            style={{ padding: '7px 18px', borderRadius: '8px', fontSize: '12px', fontWeight: 600, cursor: 'pointer', border: '1px solid rgba(255,255,255,0.1)', background: 'transparent', color: '#8898aa' }}
          >
            Annuler
          </button>
          <button
            onClick={handleConfirm}
            disabled={confirming}
            style={{ padding: '7px 18px', borderRadius: '8px', fontSize: '12px', fontWeight: 600, cursor: 'pointer', border: 'none', background: '#10b981', color: 'white', opacity: confirming ? 0.5 : 1 }}
          >
            {confirming ? '...' : '✓ Confirmer'}
          </button>
        </div>
      )}
    </div>
  )
}

export default function AgentMessage({ role, content, toolData, onConfirm }: AgentMessageProps) {
  const { user } = useAuth()

  return (
    <div style={{ display: 'flex', gap: '10px', alignItems: 'flex-start', flexDirection: role === 'user' ? 'row-reverse' : 'row' }}>
      {/* Avatar */}
      {role === 'assistant' ? (
        <div style={{ width: '28px', height: '28px', borderRadius: '8px', background: 'linear-gradient(135deg, #10b981, #059669)', display: 'flex', alignItems: 'center', justifyContent: 'center', fontSize: '12px', fontWeight: 700, color: 'white', flexShrink: 0 }}>
          R
        </div>
      ) : (
        <div style={{ width: '28px', height: '28px', borderRadius: '50%', background: 'rgba(99,102,241,0.2)', display: 'flex', alignItems: 'center', justifyContent: 'center', fontSize: '10px', fontWeight: 700, color: '#818cf8', flexShrink: 0 }}>
          {user ? avatarInitials(user.display_name) : '?'}
        </div>
      )}

      {/* Content */}
      <div style={{ maxWidth: '75%', display: 'flex', flexDirection: 'column', alignItems: role === 'user' ? 'flex-end' : 'flex-start', gap: '4px' }}>
        <span style={{ fontSize: '10px', fontWeight: 600, color: role === 'assistant' ? '#10b981' : '#818cf8' }}>
          {role === 'assistant' ? 'Reem AI' : user?.display_name ?? 'Vous'}
        </span>

        {content && (
          <div style={{
            padding: '10px 14px',
            borderRadius: role === 'user' ? '12px 12px 4px 12px' : '12px 12px 12px 4px',
            backgroundColor: role === 'user' ? 'rgba(255,255,255,0.07)' : '#151a24',
            border: `1px solid ${role === 'user' ? 'rgba(255,255,255,0.12)' : 'rgba(255,255,255,0.07)'}`,
            fontSize: '13px',
            lineHeight: 1.6,
            color: '#e8edf5',
            whiteSpace: 'pre-wrap',
            wordBreak: 'break-word',
          }}>
            {content}
          </div>
        )}

        {/* Tool results */}
        {toolData?.map((td, i) => {
          if (td.type === 'data') return <DataCard key={i} data={td.result} />
          if (td.type === 'confirm') return <ConfirmCard key={i} data={td.result as { action: string; data: Record<string, unknown> }} onConfirm={onConfirm} />
          if (td.type === 'draft_email') {
            const email = td.result as { to: string; subject: string; body: string }
            return (
              <div key={i} style={{ background: '#0f1117', border: '1px solid rgba(99,102,241,0.2)', borderRadius: '10px', padding: '14px', marginTop: '6px' }}>
                <h4 style={{ fontSize: '12px', fontWeight: 600, color: '#6366f1', marginBottom: '8px' }}>✉️ Brouillon email</h4>
                <p style={{ fontSize: '11px', color: '#8898aa' }}>À : {email.to}</p>
                <p style={{ fontSize: '11px', color: '#8898aa' }}>Objet : {email.subject}</p>
                <a
                  href={`/dashboard/email?to=${encodeURIComponent(email.to)}&subject=${encodeURIComponent(email.subject)}&body=${encodeURIComponent(email.body)}`}
                  style={{ display: 'inline-block', marginTop: '10px', padding: '7px 16px', borderRadius: '8px', background: '#6366f1', color: 'white', textDecoration: 'none', fontSize: '12px', fontWeight: 500 }}
                >
                  Ouvrir dans Email
                </a>
              </div>
            )
          }
          return null
        })}
      </div>
    </div>
  )
}
```
